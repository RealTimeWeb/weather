185            public  Forecast(JsonObject json, Gson gson) {
186                    this.periodName = json.get("time").getAsJsonObject().get("startPeriodName").getAsString();
187                    this.periodTime = json.get("time").getAsJsonObject().get("startValidTime").getAsString();
188                    this.temperatureLabel = json.get("time").getAsJsonObject().get("tempLabel").getAsString();
189                    this.temperature = json.get("data").getAsJsonObject().get("temperature").getAsInt();
190                    this.probabilityOfPrecipitation = json.get("data").getAsJsonObject().get("pop").getAsInt();
191                    this.description = json.get("data").getAsJsonObject().get("weather").getAsString();
192                    this.imageUrl = json.get("data").getAsJsonObject().get("iconLink").getAsString();
193                    this.longDescription = json.get("data").getAsJsonObject().get("text").getAsString();
194            }

219            public static ArrayList<Forecast> parse(JsonObject top, Gson gson) {
220                    JsonObject rawTime = top.get("time").getAsJsonObject();
221                    JsonObject rawData = top.get("data").getAsJsonObject();
222                    
223                    String[] periodNames = gson.fromJson(rawTime.get("startPeriodName").getAsJsonArray(), String[].class);
224                    String[] periodTimes = gson.fromJson(rawTime.get("startValidTime").getAsJsonArray(), String[].class);
225                    String[] tempLabels = gson.fromJson(rawTime.get("tempLabel").getAsJsonArray(), String[].class);
226                    int[] temperatures = gson.fromJson(rawData.get("temperature").getAsJsonArray(), int[].class);
227                    // POP is sometimes reported as "null", which means 0% chance. For your inconvenience :)
228                    JsonArray pops_array = rawData.get("pop").getAsJsonArray();
229                    int[] pops = new int[pops_array.size()];
230                    for (int i = 0; i < pops.length; i+= 1) {
231                            if (pops_array.get(i).isJsonNull()) {
232                                    pops[i] = 0;
233                            } else {
234                                    pops[i] = pops_array.get(i).getAsInt();
235                            }
236                    }
237                    String[] weather = gson.fromJson(rawData.get("weather").getAsJsonArray(), String[].class);
238                    String[] icons = gson.fromJson(rawData.get("iconLink").getAsJsonArray(), String[].class);
239                    String[] text = gson.fromJson(rawData.get("text").getAsJsonArray(), String[].class);
240    
241                    ArrayList<Forecast> forecasts = new ArrayList<Forecast>();
242                    for (int i = 0; i < periodNames.length; i+= 1) {
243                            forecasts.add(new Forecast(periodNames[i], periodTimes[i], tempLabels[i], 
244                                            temperatures[i], pops[i], weather[i], icons[i], text[i]));
245                    }
246                    return forecasts;
247            }
